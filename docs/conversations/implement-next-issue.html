11	// per profile as they have uploaded themselves. Pictures beyond that
12	// allowance are demoted to locked rendering; server-sent `locked_picture`
13	// cards are always locked (and don't consume the allowance); other card
14	// kinds are never affected. Card order is preserved.
15	export function applyPhotoAllowance(
16	  cards: ProfileCard[],
17	  uploadedPhotosCount: number,
18	): AllowedCard[] {
19	  let picturesSeen = 0;
20	
21	  return cards.map((card) => {
22	    switch (card.type) {
23	      case 'picture':

25	        return { card, locked: picturesSeen > uploadedPhotosCount };
26	      case 'locked_picture':
27	        return { card, locked: true };
28	      default:
29	        return { card, locked: false };
30	    }
31	  });
32	}
